import {
  ERROR,
  REGISTER_SUCCESS,
  SET_TOKEN,
  CHECKING_TOKEN,
  LOGIN_SUCCESS
} from './Constants.js';
import { request } from '../config';
/**
  * Create a dot.
  * @param {function} dispatch - Dispacth function
  * @param {object} data - User data object
  * @return {function} - Dispatch action
*/
export function register(dispatch, data) {
  const { username, email, password, confirm_password } = data;
  request
    .post('/auth/register', { username, email, password, confirm_password })
    .then(response =>
      dispatch({
        type: REGISTER_SUCCESS,
        errors: [],
        message:
          response.data !== undefined
            ? response.data.message
            : 'Unknown response'
      })
    )
    .catch(error => {
      dispatch({
        type: ERROR,
        errors: error.response !== undefined ? error.response.data.errors : {},
        message:
          (error.response !== undefined)
            ? error.response.data.message
            : error.message
      });
    });
}
/**
  * Create a dot.
  * @param {function} dispatch - Dispacth function
  * @param {object} data - User data object
  * @return {function} - Dispatch action
*/
export function checkToken() {
  return dispatch => {
    if (
      localStorage.getItem('access_token') &&
      localStorage.getItem('authenticated')
    ) {
      dispatch({
        type: CHECKING_TOKEN
      });
      var token = localStorage.getItem('access_token');
      return dispatch({
        type: SET_TOKEN,
        data: token
      });
    }
  };
}
/**
  * Create a dot.
  * @param {function} dispatch - Dispacth function
  * @param {object} data - User data object
  * @return {function} - Dispatch a specified action
*/
export function login(dispatch, data) {
  const { email, password } = data;
  request
    .post('/auth/login', { email, password })
    .then(response =>
      dispatch({
        type: LOGIN_SUCCESS,
        errors: [],
        message:
          response.data !== undefined
            ? response.data.message
            : 'Unknown response'
      })
    )
    .catch(error => {
      dispatch({
        type: ERROR,
        errors:
          error.response !== undefined ? error.response.data.errors : {},
        message:
          error.response !== undefined
            ? error.response.data.message
            : error.message
      });
      return false;
    });
}
